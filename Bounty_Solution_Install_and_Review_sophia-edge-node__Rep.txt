#!/bin/bash

# Sophia Edge Node Installation and Testing Script
# Target: ARM Linux (Raspberry Pi 4/5 or other SBC)
# Author: Geek Hacker Mercenary
# RTC Wallet: RTC_XXXXXXXXXXXXXXXXXXXXXXXXXXXXXX

set -e

# Configuration
LOG_FILE="/var/log/sophia_edge_test.log"
TEST_DURATION=3600  # 1 hour in seconds
GITHUB_REPO="https://github.com/Scottcjn/sophia-edge-node"
ISSUE_TEMPLATE="https://api.github.com/repos/Scottcjn/sophia-edge-node/issues"

# Hardware Information Collection
echo "=== HARDWARE INFORMATION ===" > $LOG_FILE
{
    echo "Timestamp: $(date)"
    echo "Hostname: $(hostname)"
    echo "Uptime: $(uptime)"
    echo ""
    echo "--- CPU Info ---"
    cat /proc/cpuinfo | grep -E "model name|Hardware|Revision"
    echo ""
    echo "--- Memory Info ---"
    free -h
    echo ""
    echo "--- OS Info ---"
    cat /etc/os-release
    uname -a
    echo ""
    echo "--- Disk Space ---"
    df -h /
} >> $LOG_FILE

# Installation Phase
echo "=== INSTALLATION PHASE ===" >> $LOG_FILE
cd /tmp

# Clone repository
if git clone $GITHUB_REPO 2>&1; then
    echo "✓ Repository cloned successfully" >> $LOG_FILE
    cd sophia-edge-node
    
    # Check install.sh exists and is executable
    if [[ -f "install.sh" ]]; then
        chmod +x install.sh
        echo "✓ install.sh found and made executable" >> $LOG_FILE
        
        # Run installation with full logging
        echo "--- Installation Output ---" >> $LOG_FILE
        if sudo ./install.sh 2>&1 | tee -a $LOG_FILE; then
            INSTALL_STATUS="SUCCESS"
            echo "✓ install.sh completed without errors" >> $LOG_FILE
        else
            INSTALL_STATUS="FAILED"
            echo "✗ install.sh failed with errors" >> $LOG_FILE
        fi
    else
        echo "✗ install.sh not found in repository" >> $LOG_FILE
        INSTALL_STATUS="FAILED"
    fi
else
    echo "✗ Failed to clone repository" >> $LOG_FILE
    INSTALL_STATUS="FAILED"
fi

# Miner Testing Phase (only if installation succeeded)
if [[ "$INSTALL_STATUS" == "SUCCESS" ]]; then
    echo "" >> $LOG_FILE
    echo "=== MINER TESTING PHASE ===" >> $LOG_FILE
    
    # Find and start miner process
    MINER_CMD=$(find / -name "*sophia*miner*" -type f -executable 2>/dev/null | head -1)
    
    if [[ -n "$MINER_CMD" ]]; then
        echo "Found miner executable: $MINER_CMD" >> $LOG_FILE
        
        # Start miner in background with monitoring
        echo "--- Starting Miner ---" >> $LOG_FILE
        nohup $MINER_CMD 2>&1 > /tmp/sophia_miner.log &
        MINER_PID=$!
        
        # Monitor for 1 hour
        echo "Miner PID: $MINER_PID" >> $LOG_FILE
        echo "Testing for $TEST_DURATION seconds..." >> $LOG_FILE
        
        # Check connection and attestation
        for i in {1..12}; do
            sleep 300  # Check every 5 minutes
            
            echo "" >> $LOG_FILE
            echo "--- Check #$i at $(date) ---" >> $LOG_FILE
            
            # Check if process is running
            if ps -p $MINER_PID > /dev/null; then
                echo "✓ Miner process still running" >> $LOG_FILE
            else
                echo "✗ Miner process died" >> $LOG_FILE
                break
            fi
            
            # Check logs for connection/attestation
            if grep -i "connected\|attestation\|submitted" /tmp/sophia_miner.log 2>/dev/null | tail -5; then
                echo "✓ Found connection/attestation in logs" >> $LOG_FILE
            fi
            
            # Network connections check
            echo "--- Network Connections ---" >> $LOG_FILE
            netstat -tunap 2>/dev/null | grep $MINER_PID >> $LOG_FILE || true
            
            # Resource usage
            echo "--- Resource Usage ---" >> $LOG_FILE
            ps -p $MINER_PID -o %cpu,%mem,cmd >> $LOG_FILE
        done
        
        # Kill miner after test
        kill $MINER_PID 2>/dev/null || true
        
        # Analyze logs for success patterns
        echo "" >> $LOG_FILE
        echo "=== LOG ANALYSIS ===" >> $LOG_FILE
        grep -E "(error|fail|exception|connected|attestation|success)" /tmp/sophia_miner.log 2>/dev/null | tail -20 >> $LOG_FILE
        
    else
        echo "✗ Could not find miner executable" >> $LOG_FILE
    fi
fi

# RetroArch Testing (if available)
echo "" >> $LOG_FILE
echo "=== RETROARCH TESTING ===" >> $LOG_FILE
if command -v retroarch >/dev/null 2>&1; then
    echo "✓ RetroArch detected" >> $LOG_FILE
    
    # Check for achievement detection
    if find / -name "*achievement*" -type f 2>/dev/null | grep -i sophia; then
        echo "✓ Sophia achievement files detected" >> $LOG_FILE
    else
        echo "✗ No Sophia achievement files found" >> $LOG_FILE
    fi
    
    # Check RetroArch logs
    RETROARCH_LOG="$HOME/.config/retroarch/retroarch.log"
    if [[ -f "$RETROARCH_LOG" ]]; then
        echo "--- RetroArch Log Snippet ---" >> $LOG_FILE
        tail -20 "$RETROARCH_LOG" >> $LOG_FILE
    fi
else
    echo "✗ RetroArch not installed" >> $LOG_FILE
fi

# Bug Detection and Reporting
echo "" >> $LOG_FILE
echo "=== BUG DETECTION ===" >> $LOG_FILE
BUGS_FOUND=0

# Check for common issues
if [[ "$INSTALL_STATUS" == "FAILED" ]]; then
    echo "BUG #1: Installation failure" >> $LOG_FILE
    BUGS_FOUND=$((BUGS_FOUND + 1))
fi

if ! grep -q "connected\|attestation" /tmp/sophia_miner.log 2>/dev/null; then
    echo "BUG #2: Miner not connecting/submitting attestations" >> $LOG_FILE
    BUGS_FOUND=$((BUGS_FOUND + 1))
fi

# System resource issues
if dmesg | grep -i "oom\|killed" | grep -i sophia; then
    echo "BUG #3: Out of memory issues with Sophia" >> $LOG_FILE
    BUGS_FOUND=$((BUGS_FOUND + 1))
fi

# Generate GitHub issue template for each bug
if [[ $BUGS_FOUND -gt 0 ]]; then
    echo "" >> $LOG_FILE
    echo "=== GITHUB ISSUE TEMPLATES ===" >> $LOG_FILE
    
    for bug_num in $(seq 1 $BUGS_FOUND); do
        cat << EOF >> $LOG_FILE
Issue $bug_num:
Title: [BUG] Installation/Miner Issue on $(cat /etc/os-release | grep PRETTY_NAME | cut -d'"' -f2)
Hardware: $(cat /proc/cpuinfo | grep 'Hardware' | cut -d':' -f2 | tr -d ' ')
Steps to reproduce:
1. git clone $GITHUB_REPO
2. cd sophia-edge-node
3. sudo ./install.sh
4. Run miner executable
Expected: Miner connects and submits attestations
Actual: [DESCRIBE FAILURE]
Logs attached: $LOG_FILE
EOF
    done
fi

# Final Report
echo "" >> $LOG_FILE
echo "=== FINAL REPORT ===" >> $LOG_FILE
echo "Installation: $INSTALL_STATUS" >> $LOG_FILE
echo "Miner Test: $(if grep -q "connected\|attestation" /tmp/sophia_miner.log 2>/dev/null; then echo "CONNECTED"; else echo "FAILED"; fi)" >> $LOG_FILE
echo "RetroArch: $(if command -v retroarch >/dev/null 2>&1; then echo "DETECTED"; else echo "NOT INSTALLED"; fi)" >> $LOG_FILE
echo "Bugs Found: $BUGS_FOUND" >> $LOG_FILE
echo "Test Duration: $TEST_DURATION seconds" >> $LOG_FILE
echo "RTC Wallet: RTC_XXXXXXXXXXXXXXXXXXXXXXXXXXXXXX" >> $LOG_FILE

#
